60 def _get_allsegs_and_allkinds(self): 

61 """ 

62 Create and return allsegs and allkinds by calling underlying C code. 

63 """ 

64 allsegs = [] 

65 if self.filled: 

66 lowers, uppers = self._get_lowers_and_uppers() 

67 allkinds = [] 

68 for lower, upper in zip(lowers, uppers): 

69 segs, kinds = self.cppContourGenerator.create_filled_contour( 

70 lower, upper) 

71 allsegs.append([segs]) 

72 allkinds.append([kinds]) 

73 else: 

74 allkinds = None 

75 for level in self.levels: 

76 segs = self.cppContourGenerator.create_contour(level) 

77 allsegs.append(segs) 

78 return allsegs, allkinds

80 def _contour_args(self, args, kwargs): 

81 if self.filled: 

82 fn = 'contourf' 

83 else: 

84 fn = 'contour' 

85 tri, args, kwargs = Triangulation.get_from_args_and_kwargs(*args, 

86 **kwargs) 

87 z = np.ma.asarray(args[0]) 

88 if z.shape != tri.x.shape: 

89 raise ValueError('z array must have same length as triangulation x' 

90 ' and y arrays') 

91 

92 # z values must be finite, only need to check points that are included 

93 # in the triangulation. 

94 z_check = z[np.unique(tri.get_masked_triangles())] 

95 if np.ma.is_masked(z_check): 

96 raise ValueError('z must not contain masked points within the ' 

97 'triangulation') 

98 if not np.isfinite(z_check).all(): 

99 raise ValueError('z array must not contain non-finite values ' 

100 'within the triangulation') 

101 

102 z = np.ma.masked_invalid(z, copy=False) 

103 self.zmax = float(z_check.max()) 

104 self.zmin = float(z_check.min()) 

105 if self.logscale and self.zmin <= 0: 

106 raise ValueError('Cannot %s log of negative values.' % fn) 

107 self._process_contour_level_args(args[1:]) 

108 return (tri, z)

112Draw contour %(type)s on an unstructured triangular grid. 

113 

114The triangulation can be specified in one of two ways; either :: 

115 

116 %(func)s(triangulation, ...) 

117 

118where *triangulation* is a `.Triangulation` object, or :: 

119 

120 %(func)s(x, y, ...) 

121 %(func)s(x, y, triangles, ...) 

122 %(func)s(x, y, triangles=triangles, ...) 

123 %(func)s(x, y, mask=mask, ...) 

124 %(func)s(x, y, triangles, mask=mask, ...) 

125 

126in which case a `.Triangulation` object will be created. See that class' 

127docstring for an explanation of these cases. 

128 

129The remaining arguments may be:: 

130 

131 %(func)s(..., Z) 

132 

133where *Z* is the array of values to contour, one per point in the 

134triangulation. The level values are chosen automatically. 

135 

136:: 

137 

138 %(func)s(..., Z, levels) 

139 

140contour up to *levels+1* automatically chosen contour levels (*levels* 

141intervals). 

142 

143:: 

144 

145 %(func)s(..., Z, levels) 

146 

147draw contour %(type)s at the values specified in sequence *levels*, which must 

148be in increasing order. 

149 

150:: 

151 

152 %(func)s(Z, **kwargs) 

153 

154Use keyword arguments to control colors, linewidth, origin, cmap ... see below 

155for more details. 

156 

157Parameters 

158---------- 

159triangulation : `.Triangulation`, optional 

160 The unstructured triangular grid. 

161 

162 If specified, then *x*, *y*, *triangles*, and *mask* are not accepted. 

163 

164x, y : array-like, optional 

165 The coordinates of the values in *Z*. 

166 

167triangles : int array-like of shape (ntri, 3), optional 

168 For each triangle, the indices of the three points that make up the 

169 triangle, ordered in an anticlockwise manner. If not specified, the 

170 Delaunay triangulation is calculated. 

171 

172mask : bool array-like of shape (ntri), optional 

173 Which triangles are masked out. 

174 

175Z : array-like(N, M) 

176 The height values over which the contour is drawn. 

177 

178levels : int or array-like, optional

180 

181 If an int *n*, use `~matplotlib.ticker.MaxNLocator`, which tries to 

182 automatically choose no more than *n+1* "nice" contour levels between 

183 *vmin* and *vmax*. 

184

186 be in increasing order. 

187 

188Returns 

189------- 

190`~matplotlib.tri.TriContourSet` 

191 

192Other Parameters 

193---------------- 

194colors : color string or sequence of colors, optional 

195 The colors of the levels, i.e., the contour %(type)s. 

196 

197 The sequence is cycled for the levels in ascending order. If the sequence 

198 is shorter than the number of levels, it's repeated. 

199 

200 As a shortcut, single color strings may be used in place of one-element 

201 lists, i.e. ``'red'`` instead of ``['red']`` to color all levels with the 

202 same color. This shortcut does only work for color strings, not for other 

203 ways of specifying colors. 

204

87@cbook.deprecated("3.2") 

88def apply_window(x, window, axis=0, return_window=None): 

89 """ 

90 Apply the given window to the given 1D or 2D array along the given axis. 

91 

92 Parameters 

93 ---------- 

94 x : 1D or 2D array or sequence 

95 Array or sequence containing the data. 

96 

97 window : function or array. 

98 Either a function to generate a window or an array with length 

99 *x*.shape[*axis*] 

100 

101 axis : int 

102 The axis over which to do the repetition. 

103 Must be 0 or 1. The default is 0 

104 

105 return_window : bool 

106 If true, also return the 1D values of the window that was applied 

107 """ 

108 x = np.asarray(x) 

109 

110 if x.ndim < 1 or x.ndim > 2: 

111 raise ValueError('only 1D or 2D arrays can be used') 

112 if axis+1 > x.ndim: 

113 raise ValueError('axis(=%s) out of bounds' % axis) 

114 

115 xshape = list(x.shape) 

116 xshapetarg = xshape.pop(axis) 

117 

118 if np.iterable(window): 

119 if len(window) != xshapetarg: 

120 raise ValueError('The len(window) must be the same as the shape ' 

121 'of x for the chosen axis') 

122 windowVals = window 

123 else: 

124 windowVals = window(np.ones(xshapetarg, dtype=x.dtype)) 

125 

126 if x.ndim == 1: 

127 if return_window: 

128 return windowVals * x, windowVals 

129 else: 

130 return windowVals * x 

131 

132 xshapeother = xshape.pop() 

133 

134 otheraxis = (axis+1) % 2 

135 

136 windowValsRep = stride_repeat(windowVals, xshapeother, axis=otheraxis) 

137 

138 if return_window: 

139 return windowValsRep * x, windowVals 

140 else: 

141 return windowValsRep * x

144def detrend(x, key=None, axis=None): 

145 """ 

146 Return x with its trend removed. 

147 

148 Parameters 

149 ---------- 

150 x : array or sequence 

151 Array or sequence containing the data. 

152 

153 key : {'default', 'constant', 'mean', 'linear', 'none'} or function 

154 The detrending algorithm to use. 'default', 'mean', and 'constant' are 

155 the same as `detrend_mean`. 'linear' is the same as `detrend_linear`. 

156 'none' is the same as `detrend_none`. The default is 'mean'. See the 

157 corresponding functions for more details regarding the algorithms. Can 

158 also be a function that carries out the detrend operation. 

159 

160 axis : int 

161 The axis along which to do the detrending. 

162 

163 See Also 

164 -------- 

165 detrend_mean : Implementation of the 'mean' algorithm. 

166 detrend_linear : Implementation of the 'linear' algorithm. 

167 detrend_none : Implementation of the 'none' algorithm. 

168 """ 

169 if key is None or key in ['constant', 'mean', 'default']: 

170 return detrend(x, key=detrend_mean, axis=axis) 

171 elif key == 'linear': 

172 return detrend(x, key=detrend_linear, axis=axis) 

173 elif key == 'none': 

174 return detrend(x, key=detrend_none, axis=axis) 

175 elif callable(key): 

176 x = np.asarray(x) 

177 if axis is not None and axis + 1 > x.ndim: 

178 raise ValueError(f'axis(={axis}) out of bounds') 

179 if (axis is None and x.ndim == 0) or (not axis and x.ndim == 1): 

180 return key(x) 

181 # try to use the 'axis' argument if the function supports it, 

182 # otherwise use apply_along_axis to do it 

183 try: 

184 return key(x, axis=axis) 

185 except TypeError: 

186 return np.apply_along_axis(key, axis=axis, arr=x) 

187 else: 

188 raise ValueError( 

189 f"Unknown value for key: {key!r}, must be one of: 'default', " 

190 f"'constant', 'mean', 'linear', or a function")

243def detrend_linear(y): 

244 """ 

245 Return x minus best fit line; 'linear' detrending. 

246 

247 Parameters 

248 ---------- 

249 y : 0-D or 1-D array or sequence 

250 Array or sequence containing the data 

251 

252 axis : int 

253 The axis along which to take the mean. See numpy.mean for a 

254 description of this argument. 

255 

256 See Also 

257 -------- 

258 detrend_mean : Another detrend algorithm. 

259 detrend_none : Another detrend algorithm. 

260 detrend : A wrapper around all the detrend algorithms. 

261 """ 

262 # This is faster than an algorithm based on linalg.lstsq. 

263 y = np.asarray(y) 

264 

265 if y.ndim > 1: 

266 raise ValueError('y cannot have ndim > 1') 

267 

268 # short-circuit 0-D array. 

269 if not y.ndim: 

270 return np.array(0., dtype=y.dtype) 

271 

272 x = np.arange(y.size, dtype=float) 

273 

274 C = np.cov(x, y, bias=1) 

275 b = C[0, 1]/C[0, 0] 

276 

277 a = y.mean() - b*x.mean() 

278 return y - (b*x + a)

281def stride_windows(x, n, noverlap=None, axis=0): 

282 """ 

283 Get all windows of x with length n as a single array, 

284 using strides to avoid data duplication. 

285 

286 .. warning:: 

287 

288 It is not safe to write to the output array. Multiple 

289 elements may point to the same piece of memory, 

290 so modifying one value may change others. 

291 

292 Parameters 

293 ---------- 

294 x : 1D array or sequence 

295 Array or sequence containing the data. 

296 

297 n : int 

298 The number of data points in each window. 

299 

300 noverlap : int 

301 The overlap between adjacent windows. 

302 Default is 0 (no overlap) 

303 

304 axis : int 

305 The axis along which the windows will run. 

306 

307 References 

308 ---------- 

309 `stackoverflow: Rolling window for 1D arrays in Numpy? 

310 <http://stackoverflow.com/a/6811241>`_ 

311 `stackoverflow: Using strides for an efficient moving average filter 

312 <http://stackoverflow.com/a/4947453>`_ 

313 """ 

314 if noverlap is None: 

315 noverlap = 0 

316 

317 if noverlap >= n: 

318 raise ValueError('noverlap must be less than n') 

319 if n < 1: 

320 raise ValueError('n cannot be less than 1') 

321 

322 x = np.asarray(x) 

323 

324 if x.ndim != 1: 

325 raise ValueError('only 1-dimensional arrays can be used') 

326 if n == 1 and noverlap == 0: 

327 if axis == 0: 

328 return x[np.newaxis] 

329 else: 

330 return x[np.newaxis].transpose() 

331 if n > x.size: 

332 raise ValueError('n cannot be greater than the length of x') 

333 

334 # np.lib.stride_tricks.as_strided easily leads to memory corruption for 

335 # non integer shape and strides, i.e. noverlap or n. See #3845. 

336 noverlap = int(noverlap) 

337 n = int(n) 

338 

339 step = n - noverlap 

340 if axis == 0: 

341 shape = (n, (x.shape[-1]-noverlap)//step) 

342 strides = (x.strides[0], step*x.strides[0]) 

343 else: 

344 shape = ((x.shape[-1]-noverlap)//step, n) 

345 strides = (step*x.strides[0], x.strides[0]) 

346 return np.lib.stride_tricks.as_strided(x, shape=shape, strides=strides)

349@cbook.deprecated("3.2") 

350def stride_repeat(x, n, axis=0): 

351 """ 

352 Repeat the values in an array in a memory-efficient manner. Array x is 

353 stacked vertically n times. 

354 

355 .. warning:: 

356 

357 It is not safe to write to the output array. Multiple 

358 elements may point to the same piece of memory, so 

359 modifying one value may change others. 

360 

361 Parameters 

362 ---------- 

363 x : 1D array or sequence 

364 Array or sequence containing the data. 

365 

366 n : int 

367 The number of time to repeat the array. 

368 

369 axis : int 

370 The axis along which the data will run. 

371 

372 References 

373 ---------- 

374 `stackoverflow: Repeat NumPy array without replicating data? 

375 <http://stackoverflow.com/a/5568169>`_ 

376 """ 

377 if axis not in [0, 1]: 

378 raise ValueError('axis must be 0 or 1') 

379 x = np.asarray(x) 

380 if x.ndim != 1: 

381 raise ValueError('only 1-dimensional arrays can be used') 

382 

383 if n == 1: 

384 if axis == 0: 

385 return np.atleast_2d(x) 

386 else: 

387 return np.atleast_2d(x).T 

388 if n < 1: 

389 raise ValueError('n cannot be less than 1') 

390 

391 # np.lib.stride_tricks.as_strided easily leads to memory corruption for 

392 # non integer shape and strides, i.e. n. See #3845. 

393 n = int(n) 

394 

395 if axis == 0: 

396 shape = (n, x.size) 

397 strides = (0, x.strides[0]) 

398 else: 

399 shape = (x.size, n) 

400 strides = (x.strides[0], 0) 

401 

402 return np.lib.stride_tricks.as_strided(x, shape=shape, strides=strides)

405def _spectral_helper(x, y=None, NFFT=None, Fs=None, detrend_func=None, 

406 window=None, noverlap=None, pad_to=None, 

407 sides=None, scale_by_freq=None, mode=None): 

408 """ 

409 Private helper implementing the common parts between the psd, csd, 

410 spectrogram and complex, magnitude, angle, and phase spectrums. 

411 """ 

412 if y is None: 

413 # if y is None use x for y 

414 same_data = True 

415 else: 

416 # The checks for if y is x are so that we can use the same function to 

417 # implement the core of psd(), csd(), and spectrogram() without doing 

418 # extra calculations. We return the unaveraged Pxy, freqs, and t. 

419 same_data = y is x 

420 

421 if Fs is None: 

422 Fs = 2 

423 if noverlap is None: 

424 noverlap = 0 

425 if detrend_func is None: 

426 detrend_func = detrend_none 

427 if window is None: 

428 window = window_hanning 

429 

430 # if NFFT is set to None use the whole signal 

431 if NFFT is None: 

432 NFFT = 256 

433 

434 if mode is None or mode == 'default': 

435 mode = 'psd' 

436 cbook._check_in_list( 

437 ['default', 'psd', 'complex', 'magnitude', 'angle', 'phase'], 

438 mode=mode) 

439 

440 if not same_data and mode != 'psd': 

441 raise ValueError("x and y must be equal if mode is not 'psd'") 

442 

443 # Make sure we're dealing with a numpy array. If y and x were the same 

444 # object to start with, keep them that way 

445 x = np.asarray(x) 

446 if not same_data: 

447 y = np.asarray(y) 

448 

449 if sides is None or sides == 'default': 

450 if np.iscomplexobj(x): 

451 sides = 'twosided' 

452 else: 

453 sides = 'onesided' 

454 cbook._check_in_list(['default', 'onesided', 'twosided'], sides=sides) 

455 

456 # zero pad x and y up to NFFT if they are shorter than NFFT 

457 if len(x) < NFFT: 

458 n = len(x) 

459 x = np.resize(x, NFFT) 

460 x[n:] = 0 

461 

462 if not same_data and len(y) < NFFT: 

463 n = len(y) 

464 y = np.resize(y, NFFT) 

465 y[n:] = 0 

466 

467 if pad_to is None: 

468 pad_to = NFFT 

469 

470 if mode != 'psd': 

471 scale_by_freq = False 

472 elif scale_by_freq is None: 

473 scale_by_freq = True 

474 

475 # For real x, ignore the negative frequencies unless told otherwise 

476 if sides == 'twosided': 

477 numFreqs = pad_to 

478 if pad_to % 2: 

479 freqcenter = (pad_to - 1)//2 + 1 

480 else: 

481 freqcenter = pad_to//2 

482 scaling_factor = 1. 

483 elif sides == 'onesided': 

484 if pad_to % 2: 

485 numFreqs = (pad_to + 1)//2 

486 else: 

487 numFreqs = pad_to//2 + 1 

488 scaling_factor = 2. 

489 

490 if not np.iterable(window): 

491 window = window(np.ones(NFFT, x.dtype)) 

492 if len(window) != NFFT: 

493 raise ValueError( 

494 "The window length must match the data's first dimension") 

495 

496 result = stride_windows(x, NFFT, noverlap, axis=0) 

497 result = detrend(result, detrend_func, axis=0) 

498 result = result * window.reshape((-1, 1)) 

499 result = np.fft.fft(result, n=pad_to, axis=0)[:numFreqs, :] 

500 freqs = np.fft.fftfreq(pad_to, 1/Fs)[:numFreqs] 

501 

502 if not same_data: 

503 # if same_data is False, mode must be 'psd' 

504 resultY = stride_windows(y, NFFT, noverlap) 

505 resultY = detrend(resultY, detrend_func, axis=0) 

506 resultY = resultY * window.reshape((-1, 1)) 

507 resultY = np.fft.fft(resultY, n=pad_to, axis=0)[:numFreqs, :] 

508 result = np.conj(result) * resultY 

509 elif mode == 'psd': 

510 result = np.conj(result) * result 

511 elif mode == 'magnitude': 

512 result = np.abs(result) / np.abs(window).sum() 

513 elif mode == 'angle' or mode == 'phase': 

514 # we unwrap the phase later to handle the onesided vs. twosided case 

515 result = np.angle(result) 

516 elif mode == 'complex': 

517 result /= np.abs(window).sum() 

518 

519 if mode == 'psd': 

520 

521 # Also include scaling factors for one-sided densities and dividing by 

522 # the sampling frequency, if desired. Scale everything, except the DC 

523 # component and the NFFT/2 component: 

524 

525 # if we have a even number of frequencies, don't scale NFFT/2 

526 if not NFFT % 2: 

527 slc = slice(1, -1, None) 

528 # if we have an odd number, just don't scale DC 

529 else: 

530 slc = slice(1, None, None) 

531 

532 result[slc] *= scaling_factor 

533 

534 # MATLAB divides by the sampling frequency so that density function 

535 # has units of dB/Hz and can be integrated by the plotted frequency 

536 # values. Perform the same scaling here. 

537 if scale_by_freq: 

538 result /= Fs 

539 # Scale the spectrum by the norm of the window to compensate for 

540 # windowing loss; see Bendat & Piersol Sec 11.5.2. 

541 result /= (np.abs(window)**2).sum() 

542 else: 

543 # In this case, preserve power in the segment, not amplitude 

544 result /= np.abs(window).sum()**2 

545 

546 t = np.arange(NFFT/2, len(x) - NFFT/2 + 1, NFFT - noverlap)/Fs 

547 

548 if sides == 'twosided': 

549 # center the frequency range at zero 

550 freqs = np.roll(freqs, -freqcenter, axis=0) 

551 result = np.roll(result, -freqcenter, axis=0) 

552 elif not pad_to % 2: 

553 # get the last value correctly, it is negative otherwise 

554 freqs[-1] *= -1 

555 

556 # we unwrap the phase here to handle the onesided vs. twosided case 

557 if mode == 'phase': 

558 result = np.unwrap(result, axis=0) 

559 

560 return result, freqs, t

564 mode, x, Fs=None, window=None, pad_to=None, sides=None): 

565 """ 

566 Private helper implementing the commonality between the complex, magnitude, 

567 angle, and phase spectrums.

569 cbook._check_in_list(['complex', 'magnitude', 'angle', 'phase'], mode=mode) 

570 

571 if pad_to is None: 

572 pad_to = len(x) 

573

575 detrend_func=detrend_none, window=window,

577 sides=sides,

582

584 spec = spec[:, 0] 

585 

586 return spec, freqs

649def psd(x, NFFT=None, Fs=None, detrend=None, window=None, 

650 noverlap=None, pad_to=None, sides=None, scale_by_freq=None): 

651 r""" 

652 Compute the power spectral density. 

653 

654 The power spectral density :math:`P_{xx}` by Welch's average 

655 periodogram method. The vector *x* is divided into *NFFT* length 

656 segments. Each segment is detrended by function *detrend* and 

657 windowed by function *window*. *noverlap* gives the length of 

658 the overlap between segments. The :math:`|\mathrm{fft}(i)|^2` 

659 of each segment :math:`i` are averaged to compute :math:`P_{xx}`. 

660 

661 If len(*x*) < *NFFT*, it will be zero padded to *NFFT*. 

662 

663 Parameters 

664 ---------- 

665 x : 1-D array or sequence 

666 Array or sequence containing the data 

667 

668 %(Spectral)s 

669 

670 %(PSD)s 

671 

672 noverlap : int 

673 The number of points of overlap between segments. 

674 The default value is 0 (no overlap). 

675 

676 Returns 

677 ------- 

678 Pxx : 1-D array 

679 The values for the power spectrum :math:`P_{xx}` (real valued) 

680 

681 freqs : 1-D array 

682 The frequencies corresponding to the elements in *Pxx* 

683 

684 References 

685 ---------- 

686 Bendat & Piersol -- Random Data: Analysis and Measurement Procedures, John 

687 Wiley & Sons (1986) 

688 

689 See Also 

690 -------- 

691 specgram

706def csd(x, y, NFFT=None, Fs=None, detrend=None, window=None, 

707 noverlap=None, pad_to=None, sides=None, scale_by_freq=None): 

708 """ 

709 Compute the cross-spectral density. 

710 

711 The cross spectral density :math:`P_{xy}` by Welch's average 

712 periodogram method. The vectors *x* and *y* are divided into 

713 *NFFT* length segments. Each segment is detrended by function 

714 *detrend* and windowed by function *window*. *noverlap* gives 

715 the length of the overlap between segments. The product of 

716 the direct FFTs of *x* and *y* are averaged over each segment 

717 to compute :math:`P_{xy}`, with a scaling to correct for power 

718 loss due to windowing. 

719 

720 If len(*x*) < *NFFT* or len(*y*) < *NFFT*, they will be zero 

721 padded to *NFFT*. 

722 

723 Parameters 

724 ---------- 

725 x, y : 1-D arrays or sequences 

726 Arrays or sequences containing the data 

727 

728 %(Spectral)s 

729 

730 %(PSD)s 

731 

732 noverlap : int 

733 The number of points of overlap between segments. 

734 The default value is 0 (no overlap). 

735 

736 Returns 

737 ------- 

738 Pxy : 1-D array 

739 The values for the cross spectrum :math:`P_{xy}` before scaling (real 

740 valued) 

741 

742 freqs : 1-D array 

743 The frequencies corresponding to the elements in *Pxy* 

744 

745 References 

746 ---------- 

747 Bendat & Piersol -- Random Data: Analysis and Measurement Procedures, John 

748 Wiley & Sons (1986) 

749 

750 See Also 

751 -------- 

752 psd : equivalent to setting ``y = x``. 

753 """ 

754 if NFFT is None: 

755 NFFT = 256 

756 Pxy, freqs, _ = _spectral_helper(x=x, y=y, NFFT=NFFT, Fs=Fs, 

757 detrend_func=detrend, window=window, 

758 noverlap=noverlap, pad_to=pad_to, 

759 sides=sides, scale_by_freq=scale_by_freq, 

760 mode='psd') 

761 

762 if Pxy.ndim == 2: 

763 if Pxy.shape[1] > 1: 

764 Pxy = Pxy.mean(axis=1) 

765 else: 

766 Pxy = Pxy[:, 0] 

767 return Pxy, freqs

912def cohere(x, y, NFFT=256, Fs=2, detrend=detrend_none, window=window_hanning, 

913 noverlap=0, pad_to=None, sides='default', scale_by_freq=None): 

914 r""" 

915 The coherence between *x* and *y*. Coherence is the normalized 

916 cross spectral density: 

917 

918 .. math:: 

919 

920 C_{xy} = \frac{|P_{xy}|^2}{P_{xx}P_{yy}} 

921 

922 Parameters 

923 ---------- 

924 x, y 

925 Array or sequence containing the data 

926 

927 %(Spectral)s 

928 

929 %(PSD)s 

930 

931 noverlap : int 

932 The number of points of overlap between blocks. The default value 

933 is 0 (no overlap). 

934 

935 Returns 

936 ------- 

937 The return value is the tuple (*Cxy*, *f*), where *f* are the 

938 frequencies of the coherence vector. For cohere, scaling the 

939 individual densities by the sampling frequency has no effect, 

940 since the factors cancel out. 

941 

942 See Also 

943 -------- 

944 :func:`psd`, :func:`csd` : 

945 For information about the methods used to compute :math:`P_{xy}`, 

946 :math:`P_{xx}` and :math:`P_{yy}`. 

947 """ 

948 if len(x) < 2 * NFFT: 

949 raise ValueError( 

950 "Coherence is calculated by averaging over *NFFT* length " 

951 "segments. Your signal is too short for your choice of *NFFT*.") 

952 Pxx, f = psd(x, NFFT, Fs, detrend, window, noverlap, pad_to, sides, 

953 scale_by_freq) 

954 Pyy, f = psd(y, NFFT, Fs, detrend, window, noverlap, pad_to, sides, 

955 scale_by_freq) 

956 Pxy, f = csd(x, y, NFFT, Fs, detrend, window, noverlap, pad_to, sides, 

957 scale_by_freq) 

958 Cxy = np.abs(Pxy) ** 2 / (Pxx * Pyy) 

959 return Cxy, f

962class GaussianKDE: 

963 """ 

964 Representation of a kernel-density estimate using Gaussian kernels. 

965 

966 Parameters 

967 ---------- 

968 dataset : array-like 

969 Datapoints to estimate from. In case of univariate data this is a 1-D 

970 array, otherwise a 2-D array with shape (# of dims, # of data). 

971 

972 bw_method : str, scalar or callable, optional 

973 The method used to calculate the estimator bandwidth. This can be 

974 'scott', 'silverman', a scalar constant or a callable. If a 

975 scalar, this will be used directly as `kde.factor`. If a 

976 callable, it should take a `GaussianKDE` instance as only 

977 parameter and return a scalar. If None (default), 'scott' is used. 

978 

979 Attributes 

980 ---------- 

981 dataset : ndarray 

982 The dataset with which `gaussian_kde` was initialized. 

983 

984 dim : int 

985 Number of dimensions. 

986 

987 num_dp : int 

988 Number of datapoints. 

989 

990 factor : float 

991 The bandwidth factor, obtained from `kde.covariance_factor`, with which 

992 the covariance matrix is multiplied. 

993 

994 covariance : ndarray 

995 The covariance matrix of *dataset*, scaled by the calculated bandwidth 

996 (`kde.factor`). 

997 

998 inv_cov : ndarray 

999 The inverse of *covariance*. 

1000 

1001 Methods 

1002 ------- 

1003 kde.evaluate(points) : ndarray 

1004 Evaluate the estimated pdf on a provided set of points. 

1005 

1006 kde(points) : ndarray 

1007 Same as kde.evaluate(points) 

1008 

1009 """ 

1010 

1011 # This implementation with minor modification was too good to pass up. 

1012 # from scipy: https://github.com/scipy/scipy/blob/master/scipy/stats/kde.py 

1013 

1014 def __init__(self, dataset, bw_method=None): 

1015 self.dataset = np.atleast_2d(dataset) 

1016 if not np.array(self.dataset).size > 1: 

1017 raise ValueError("`dataset` input should have multiple elements.") 

1018 

1019 self.dim, self.num_dp = np.array(self.dataset).shape 

1020 

1021 if bw_method is None: 

1022 pass 

1023 elif cbook._str_equal(bw_method, 'scott'): 

1024 self.covariance_factor = self.scotts_factor 

1025 elif cbook._str_equal(bw_method, 'silverman'): 

1026 self.covariance_factor = self.silverman_factor 

1027 elif isinstance(bw_method, Number): 

1028 self._bw_method = 'use constant' 

1029 self.covariance_factor = lambda: bw_method 

1030 elif callable(bw_method): 

1031 self._bw_method = bw_method 

1032 self.covariance_factor = lambda: self._bw_method(self) 

1033 else: 

1034 raise ValueError("`bw_method` should be 'scott', 'silverman', a " 

1035 "scalar or a callable") 

1036 

1037 # Computes the covariance matrix for each Gaussian kernel using 

1038 # covariance_factor(). 

1039 

1040 self.factor = self.covariance_factor() 

1041 # Cache covariance and inverse covariance of the data 

1042 if not hasattr(self, '_data_inv_cov'): 

1043 self.data_covariance = np.atleast_2d( 

1044 np.cov( 

1045 self.dataset, 

1046 rowvar=1, 

1047 bias=False)) 

1048 self.data_inv_cov = np.linalg.inv(self.data_covariance) 

1049 

1050 self.covariance = self.data_covariance * self.factor ** 2 

1051 self.inv_cov = self.data_inv_cov / self.factor ** 2 

1052 self.norm_factor = (np.sqrt(np.linalg.det(2 * np.pi * self.covariance)) 

1053 * self.num_dp) 

1054 

1055 def scotts_factor(self): 

1056 return np.power(self.num_dp, -1. / (self.dim + 4)) 

1057 

1058 def silverman_factor(self): 

1059 return np.power( 

1060 self.num_dp * (self.dim + 2.0) / 4.0, -1. / (self.dim + 4)) 

1061 

1062 # Default method to calculate bandwidth, can be overwritten by subclass 

1063 covariance_factor = scotts_factor 

1064 

1065 def evaluate(self, points): 

1066 """ 

1067 Evaluate the estimated pdf on a set of points. 

1068 

1069 Parameters 

1070 ---------- 

1071 points : (# of dimensions, # of points)-array 

1072 Alternatively, a (# of dimensions,) vector can be passed in and 

1073 treated as a single point. 

1074 

1075 Returns 

1076 ------- 

1077 (# of points,)-array 

1078 The values at each point. 

1079 

1080 Raises 

1081 ------ 

1082 ValueError : if the dimensionality of the input points is different 

1083 than the dimensionality of the KDE. 

1084 

1085 """ 

1086 points = np.atleast_2d(points) 

1087 

1088 dim, num_m = np.array(points).shape 

1089 if dim != self.dim: 

1090 raise ValueError("points have dimension {}, dataset has dimension " 

1091 "{}".format(dim, self.dim)) 

1092 

1093 result = np.zeros(num_m) 

1094 

1095 if num_m >= self.num_dp: 

1096 # there are more points than data, so loop over data 

1097 for i in range(self.num_dp): 

1098 diff = self.dataset[:, i, np.newaxis] - points 

1099 tdiff = np.dot(self.inv_cov, diff) 

1100 energy = np.sum(diff * tdiff, axis=0) / 2.0 

1101 result = result + np.exp(-energy) 

1102 else: 

1103 # loop over points 

1104 for i in range(num_m): 

1105 diff = self.dataset - points[:, i, np.newaxis] 

1106 tdiff = np.dot(self.inv_cov, diff) 

1107 energy = np.sum(diff * tdiff, axis=0) / 2.0 

1108 result[i] = np.sum(np.exp(-energy), axis=0) 

1109 

1110 result = result / self.norm_factor 

1111 

1112 return result

1""" 

2Adjust subplot layouts so that there are no overlapping axes or axes 

3decorations. All axes decorations are dealt with (labels, ticks, titles, 

4ticklabels) and some dependent artists are also dealt with (colorbar, suptitle, 

5legend). 

6 

7Layout is done via `~matplotlib.gridspec`, with one constraint per gridspec, 

8so it is possible to have overlapping axes if the gridspecs overlap (i.e. 

9using `~matplotlib.gridspec.GridSpecFromSubplotSpec`). Axes placed using 

10``figure.subplots()`` or ``figure.add_subplots()`` will participate in the 

11layout. Axes manually placed via ``figure.add_axes()`` will not. 

12 

13See Tutorial: :doc:`/tutorials/intermediate/constrainedlayout_guide` 

14"""

60def _axes_all_finite_sized(fig): 

61 """Return whether all axes in the figure have a finite width and height.""" 

62 for ax in fig.axes: 

63 if ax._layoutbox is not None: 

64 newpos = ax._poslayoutbox.get_rect() 

65 if newpos[2] <= 0 or newpos[3] <= 0: 

66 return False 

67 return True

71def do_constrained_layout(fig, renderer, h_pad, w_pad, 

72 hspace=None, wspace=None): 

73 """ 

74 Do the constrained_layout. Called at draw time in 

75 ``figure.constrained_layout()`` 

76 

77 Parameters 

78 ---------- 

79 fig : Figure 

80 is the ``figure`` instance to do the layout in. 

81 

82 renderer : Renderer 

83 the renderer to use. 

84 

85 h_pad, w_pad : float 

86 are in figure-normalized units, and are a padding around the axes 

87 elements. 

88 

89 hspace, wspace : float 

90 are in fractions of the subplot sizes. 

91 

92 """ 

93 

94 # Steps: 

95 # 

96 # 1. get a list of unique gridspecs in this figure. Each gridspec will be 

97 # constrained separately. 

98 # 2. Check for gaps in the gridspecs. i.e. if not every axes slot in the 

99 # gridspec has been filled. If empty, add a ghost axis that is made so 

100 # that it cannot be seen (though visible=True). This is needed to make 

101 # a blank spot in the layout. 

102 # 3. Compare the tight_bbox of each axes to its `position`, and assume that 

103 # the difference is the space needed by the elements around the edge of 

104 # the axes (decorations) like the title, ticklabels, x-labels, etc. This 

105 # can include legends who overspill the axes boundaries. 

106 # 4. Constrain gridspec elements to line up: 

107 # a) if colnum0 != colnumC, the two subplotspecs are stacked next to 

108 # each other, with the appropriate order. 

109 # b) if colnum0 == colnumC, line up the left or right side of the 

110 # _poslayoutbox (depending if it is the min or max num that is equal). 

111 # c) do the same for rows... 

112 # 5. The above doesn't constrain relative sizes of the _poslayoutboxes 

113 # at all, and indeed zero-size is a solution that the solver often finds 

114 # more convenient than expanding the sizes. Right now the solution is to 

115 # compare subplotspec sizes (i.e. drowsC and drows0) and constrain the 

116 # larger _poslayoutbox to be larger than the ratio of the sizes. i.e. if 

117 # drows0 > drowsC, then ax._poslayoutbox > axc._poslayoutbox*drowsC/drows0. 

118 # This works fine *if* the decorations are similar between the axes. 

119 # If the larger subplotspec has much larger axes decorations, then the 

120 # constraint above is incorrect. 

121 # 

122 # We need the greater than in the above, in general, rather than an equals 

123 # sign. Consider the case of the left column having 2 rows, and the right 

124 # column having 1 row. We want the top and bottom of the _poslayoutboxes 

125 # to line up. So that means if there are decorations on the left column 

126 # axes they will be smaller than half as large as the right hand axis. 

127 # 

128 # This can break down if the decoration size for the right hand axis (the 

129 # margins) is very large. There must be a math way to check for this case. 

130 

131 invTransFig = fig.transFigure.inverted().transform_bbox 

132 

133 # list of unique gridspecs that contain child axes: 

134 gss = set() 

135 for ax in fig.axes: 

136 if hasattr(ax, 'get_subplotspec'): 

137 gs = ax.get_subplotspec().get_gridspec() 

138 if gs._layoutbox is not None: 

139 gss.add(gs) 

140 if len(gss) == 0: 

141 cbook._warn_external('There are no gridspecs with layoutboxes. ' 

142 'Possibly did not call parent GridSpec with the' 

143 ' figure= keyword') 

144 

145 if fig._layoutbox.constrained_layout_called < 1: 

146 for gs in gss: 

147 # fill in any empty gridspec slots w/ ghost axes... 

148 _make_ghost_gridspec_slots(fig, gs) 

149 

150 for _ in range(2): 

151 # do the algorithm twice. This has to be done because decorators 

152 # change size after the first re-position (i.e. x/yticklabels get 

153 # larger/smaller). This second reposition tends to be much milder, 

154 # so doing twice makes things work OK. 

155 for ax in fig.axes: 

156 _log.debug(ax._layoutbox) 

157 if ax._layoutbox is not None: 

158 # make margins for each layout box based on the size of 

159 # the decorators. 

160 _make_layout_margins(ax, renderer, h_pad, w_pad) 

161 

162 # do layout for suptitle. 

163 suptitle = fig._suptitle 

164 do_suptitle = (suptitle is not None and 

165 suptitle._layoutbox is not None and 

166 suptitle.get_in_layout()) 

167 if do_suptitle: 

168 bbox = invTransFig( 

169 suptitle.get_window_extent(renderer=renderer)) 

170 height = bbox.height 

171 if np.isfinite(height): 

172 # reserve at top of figure include an h_pad above and below 

173 suptitle._layoutbox.edit_height(height + h_pad * 2) 

174 

175 # OK, the above lines up ax._poslayoutbox with ax._layoutbox 

176 # now we need to 

177 # 1) arrange the subplotspecs. We do it at this level because 

178 # the subplotspecs are meant to contain other dependent axes 

179 # like colorbars or legends. 

180 # 2) line up the right and left side of the ax._poslayoutbox 

181 # that have the same subplotspec maxes. 

182 

183 if fig._layoutbox.constrained_layout_called < 1: 

184 # arrange the subplotspecs... This is all done relative to each 

185 # other. Some subplotspecs contain axes, and others contain 

186 # gridspecs the ones that contain gridspecs are a set proportion 

187 # of their parent gridspec. The ones that contain axes are 

188 # not so constrained. 

189 figlb = fig._layoutbox 

190 for child in figlb.children: 

191 if child._is_gridspec_layoutbox(): 

192 # This routine makes all the subplot spec containers 

193 # have the correct arrangement. It just stacks the 

194 # subplot layoutboxes in the correct order... 

195 _arrange_subplotspecs(child, hspace=hspace, wspace=wspace) 

196 

197 for gs in gss: 

198 _align_spines(fig, gs) 

199 

200 fig._layoutbox.constrained_layout_called += 1 

201 fig._layoutbox.update_variables() 

202 

203 # check if any axes collapsed to zero. If not, don't change positions: 

204 if _axes_all_finite_sized(fig): 

205 # Now set the position of the axes... 

206 for ax in fig.axes: 

207 if ax._layoutbox is not None: 

208 newpos = ax._poslayoutbox.get_rect() 

209 # Now set the new position. 

210 # ax.set_position will zero out the layout for 

211 # this axis, allowing users to hard-code the position, 

212 # so this does the same w/o zeroing layout. 

213 ax._set_position(newpos, which='original') 

214 if do_suptitle: 

215 newpos = suptitle._layoutbox.get_rect() 

216 suptitle.set_y(1.0 - h_pad) 

217 else: 

218 if suptitle is not None and suptitle._layoutbox is not None: 

219 suptitle._layoutbox.edit_height(0) 

220 else: 

221 cbook._warn_external('constrained_layout not applied. At least ' 

222 'one axes collapsed to zero width or height.')

225def _make_ghost_gridspec_slots(fig, gs): 

226 """ 

227 Check for unoccupied gridspec slots and make ghost axes for these 

228 slots... Do for each gs separately. This is a pretty big kludge 

229 but shouldn't have too much ill effect. The worst is that 

230 someone querying the figure will wonder why there are more 

231 axes than they thought. 

232 """ 

233 nrows, ncols = gs.get_geometry() 

234 hassubplotspec = np.zeros(nrows * ncols, dtype=bool) 

235 axs = [] 

236 for ax in fig.axes: 

237 if (hasattr(ax, 'get_subplotspec') 

238 and ax._layoutbox is not None 

239 and ax.get_subplotspec().get_gridspec() == gs): 

240 axs += [ax] 

241 for ax in axs: 

242 ss0 = ax.get_subplotspec() 

243 hassubplotspec[ss0.num1:(ss0.num2 + 1)] = True 

244 for nn, hss in enumerate(hassubplotspec): 

245 if not hss: 

246 # this gridspec slot doesn't have an axis so we 

247 # make a "ghost". 

248 ax = fig.add_subplot(gs[nn]) 

249 ax.set_visible(False)

252def _make_layout_margins(ax, renderer, h_pad, w_pad): 

253 """ 

254 For each axes, make a margin between the *pos* layoutbox and the 

255 *axes* layoutbox be a minimum size that can accommodate the 

256 decorations on the axis. 

257 """ 

258 fig = ax.figure 

259 invTransFig = fig.transFigure.inverted().transform_bbox 

260 pos = ax.get_position(original=True) 

261 try: 

262 tightbbox = ax.get_tightbbox(renderer=renderer, for_layout_only=True) 

263 except TypeError: 

264 tightbbox = ax.get_tightbbox(renderer=renderer) 

265 

266 if tightbbox is None: 

267 bbox = pos 

268 else: 

269 bbox = invTransFig(tightbbox) 

270 

271 # this can go wrong: 

272 if not (np.isfinite(bbox.width) and np.isfinite(bbox.height)): 

273 # just abort, this is likely a bad set of coordinates that 

274 # is transitory... 

275 return 

276 # use stored h_pad if it exists 

277 h_padt = ax._poslayoutbox.h_pad 

278 if h_padt is None: 

279 h_padt = h_pad 

280 w_padt = ax._poslayoutbox.w_pad 

281 if w_padt is None: 

282 w_padt = w_pad 

283 ax._poslayoutbox.edit_left_margin_min(-bbox.x0 + pos.x0 + w_padt) 

284 ax._poslayoutbox.edit_right_margin_min(bbox.x1 - pos.x1 + w_padt) 

285 ax._poslayoutbox.edit_bottom_margin_min(-bbox.y0 + pos.y0 + h_padt) 

286 ax._poslayoutbox.edit_top_margin_min(bbox.y1-pos.y1+h_padt) 

287 _log.debug('left %f', (-bbox.x0 + pos.x0 + w_pad)) 

288 _log.debug('right %f', (bbox.x1 - pos.x1 + w_pad)) 

289 _log.debug('bottom %f', (-bbox.y0 + pos.y0 + h_padt)) 

290 _log.debug('bbox.y0 %f', bbox.y0) 

291 _log.debug('pos.y0 %f', pos.y0) 

292 # Sometimes its possible for the solver to collapse 

293 # rather than expand axes, so they all have zero height 

294 # or width. This stops that... It *should* have been 

295 # taken into account w/ pref_width... 

296 if fig._layoutbox.constrained_layout_called < 1: 

297 ax._poslayoutbox.constrain_height_min(20, strength='weak') 

298 ax._poslayoutbox.constrain_width_min(20, strength='weak') 

299 ax._layoutbox.constrain_height_min(20, strength='weak') 

300 ax._layoutbox.constrain_width_min(20, strength='weak') 

301 ax._poslayoutbox.constrain_top_margin(0, strength='weak') 

302 ax._poslayoutbox.constrain_bottom_margin(0, strength='weak') 

303 ax._poslayoutbox.constrain_right_margin(0, strength='weak') 

304 ax._poslayoutbox.constrain_left_margin(0, strength='weak')

307def _align_spines(fig, gs): 

308 """ 

309 - Align right/left and bottom/top spines of appropriate subplots. 

310 - Compare size of subplotspec including height and width ratios 

311 and make sure that the axes spines are at least as large 

312 as they should be. 

313 """ 

314 # for each gridspec... 

315 nrows, ncols = gs.get_geometry() 

316 width_ratios = gs.get_width_ratios() 

317 height_ratios = gs.get_height_ratios() 

318 if width_ratios is None: 

319 width_ratios = np.ones(ncols) 

320 if height_ratios is None: 

321 height_ratios = np.ones(nrows) 

322 

323 # get axes in this gridspec.... 

324 axs = [ax for ax in fig.axes 

325 if (hasattr(ax, 'get_subplotspec') 

326 and ax._layoutbox is not None 

327 and ax.get_subplotspec().get_gridspec() == gs)] 

328 rowspans = [] 

329 colspans = [] 

330 heights = [] 

331 widths = [] 

332 

333 for ax in axs: 

334 ss0 = ax.get_subplotspec() 

335 rowspan = ss0.rowspan 

336 colspan = ss0.colspan 

337 rowspans.append(rowspan) 

338 colspans.append(colspan) 

339 heights.append(sum(height_ratios[rowspan.start:rowspan.stop])) 

340 widths.append(sum(width_ratios[colspan.start:colspan.stop])) 

341 

342 for idx0, ax0 in enumerate(axs): 

343 # Compare ax to all other axs: If the subplotspecs start (/stop) at 

344 # the same column, then line up their left (/right) sides; likewise 

345 # for rows/top/bottom. 

346 rowspan0 = rowspans[idx0] 

347 colspan0 = colspans[idx0] 

348 height0 = heights[idx0] 

349 width0 = widths[idx0] 

350 alignleft = False 

351 alignright = False 

352 alignbot = False 

353 aligntop = False 

354 alignheight = False 

355 alignwidth = False 

356 for idx1 in range(idx0 + 1, len(axs)): 

357 ax1 = axs[idx1] 

358 rowspan1 = rowspans[idx1] 

359 colspan1 = colspans[idx1] 

360 width1 = widths[idx1] 

361 height1 = heights[idx1] 

362 # Horizontally align axes spines if they have the same min or max: 

363 if not alignleft and colspan0.start == colspan1.start: 

364 _log.debug('same start columns; line up layoutbox lefts') 

365 layoutbox.align([ax0._poslayoutbox, ax1._poslayoutbox], 

366 'left') 

367 alignleft = True 

368 if not alignright and colspan0.stop == colspan1.stop: 

369 _log.debug('same stop columns; line up layoutbox rights') 

370 layoutbox.align([ax0._poslayoutbox, ax1._poslayoutbox], 

371 'right') 

372 alignright = True 

373 # Vertically align axes spines if they have the same min or max: 

374 if not aligntop and rowspan0.start == rowspan1.start: 

375 _log.debug('same start rows; line up layoutbox tops') 

376 layoutbox.align([ax0._poslayoutbox, ax1._poslayoutbox], 

377 'top') 

378 aligntop = True 

379 if not alignbot and rowspan0.stop == rowspan1.stop: 

380 _log.debug('same stop rows; line up layoutbox bottoms') 

381 layoutbox.align([ax0._poslayoutbox, ax1._poslayoutbox], 

382 'bottom') 

383 alignbot = True 

384 

385 # Now we make the widths and heights of position boxes 

386 # similar. (i.e the spine locations) 

387 # This allows vertically stacked subplots to have different sizes 

388 # if they occupy different amounts of the gridspec, e.g. if 

389 # gs = gridspec.GridSpec(3, 1) 

390 # ax0 = gs[0, :] 

391 # ax1 = gs[1:, :] 

392 # then len(rowspan0) = 1, and len(rowspan1) = 2, 

393 # and ax1 should be at least twice as large as ax0. 

394 # But it can be more than twice as large because 

395 # it needs less room for the labeling. 

396 

397 # For heights, do it if the subplots share a column. 

398 if not alignheight and len(rowspan0) == len(rowspan1): 

399 ax0._poslayoutbox.constrain_height( 

400 ax1._poslayoutbox.height * height0 / height1) 

401 alignheight = True 

402 elif _spans_overlap(colspan0, colspan1): 

403 if height0 > height1: 

404 ax0._poslayoutbox.constrain_height_min( 

405 ax1._poslayoutbox.height * height0 / height1) 

406 elif height0 < height1: 

407 ax1._poslayoutbox.constrain_height_min( 

408 ax0._poslayoutbox.height * height1 / height0) 

409 # For widths, do it if the subplots share a row. 

410 if not alignwidth and len(colspan0) == len(colspan1): 

411 ax0._poslayoutbox.constrain_width( 

412 ax1._poslayoutbox.width * width0 / width1) 

413 alignwidth = True 

414 elif _spans_overlap(rowspan0, rowspan1): 

415 if width0 > width1: 

416 ax0._poslayoutbox.constrain_width_min( 

417 ax1._poslayoutbox.width * width0 / width1) 

418 elif width0 < width1: 

419 ax1._poslayoutbox.constrain_width_min( 

420 ax0._poslayoutbox.width * width1 / width0)

423def _arrange_subplotspecs(gs, hspace=0, wspace=0): 

424 """Recursively arrange the subplotspec children of the given gridspec.""" 

425 sschildren = [] 

426 for child in gs.children: 

427 if child._is_subplotspec_layoutbox(): 

428 for child2 in child.children: 

429 # check for gridspec children... 

430 if child2._is_gridspec_layoutbox(): 

431 _arrange_subplotspecs(child2, hspace=hspace, wspace=wspace) 

432 sschildren += [child] 

433 # now arrange the subplots... 

434 for child0 in sschildren: 

435 ss0 = child0.artist 

436 nrows, ncols = ss0.get_gridspec().get_geometry() 

437 rowspan0 = ss0.rowspan 

438 colspan0 = ss0.colspan 

439 sschildren = sschildren[1:] 

440 for child1 in sschildren: 

441 ss1 = child1.artist 

442 rowspan1 = ss1.rowspan 

443 colspan1 = ss1.colspan 

444 # OK, this tells us the relative layout of child0 with child1. 

445 pad = wspace / ncols 

446 if colspan0.stop <= colspan1.start: 

447 layoutbox.hstack([ss0._layoutbox, ss1._layoutbox], padding=pad) 

448 if colspan1.stop <= colspan0.start: 

449 layoutbox.hstack([ss1._layoutbox, ss0._layoutbox], padding=pad) 

450 # vertical alignment 

451 pad = hspace / nrows 

452 if rowspan0.stop <= rowspan1.start: 

453 layoutbox.vstack([ss0._layoutbox, ss1._layoutbox], padding=pad) 

454 if rowspan1.stop <= rowspan0.start: 

455 layoutbox.vstack([ss1._layoutbox, ss0._layoutbox], padding=pad)

458def layoutcolorbarsingle(ax, cax, shrink, aspect, location, pad=0.05): 

459 """ 

460 Do the layout for a colorbar, to not overly pollute colorbar.py 

461 

462 *pad* is in fraction of the original axis size. 

463 """ 

464 axlb = ax._layoutbox 

465 axpos = ax._poslayoutbox 

466 axsslb = ax.get_subplotspec()._layoutbox 

467 lb = layoutbox.LayoutBox( 

468 parent=axsslb, 

469 name=axsslb.name + '.cbar', 

470 artist=cax) 

471 

472 if location in ('left', 'right'): 

473 lbpos = layoutbox.LayoutBox( 

474 parent=lb, 

475 name=lb.name + '.pos', 

476 tightwidth=False, 

477 pos=True, 

478 subplot=False, 

479 artist=cax) 

480 

481 if location == 'right': 

482 # arrange to right of parent axis 

483 layoutbox.hstack([axlb, lb], padding=pad * axlb.width, 

484 strength='strong') 

485 else: 

486 layoutbox.hstack([lb, axlb], padding=pad * axlb.width) 

487 # constrain the height and center... 

488 layoutbox.match_heights([axpos, lbpos], [1, shrink]) 

489 layoutbox.align([axpos, lbpos], 'v_center') 

490 # set the width of the pos box 

491 lbpos.constrain_width(shrink * axpos.height * (1/aspect), 

492 strength='strong') 

493 elif location in ('bottom', 'top'): 

494 lbpos = layoutbox.LayoutBox( 

495 parent=lb, 

496 name=lb.name + '.pos', 

497 tightheight=True, 

498 pos=True, 

499 subplot=False, 

500 artist=cax) 

501 

502 if location == 'bottom': 

503 layoutbox.vstack([axlb, lb], padding=pad * axlb.height) 

504 else: 

505 layoutbox.vstack([lb, axlb], padding=pad * axlb.height) 

506 # constrain the height and center... 

507 layoutbox.match_widths([axpos, lbpos], 

508 [1, shrink], strength='strong') 

509 layoutbox.align([axpos, lbpos], 'h_center') 

510 # set the height of the pos box 

511 lbpos.constrain_height(axpos.width * aspect * shrink, 

512 strength='medium') 

513 

514 return lb, lbpos

517def _getmaxminrowcolumn(axs): 

518 """ 

519 Find axes covering the first and last rows and columns of a list of axes. 

520 """ 

521 startrow = startcol = np.inf 

522 stoprow = stopcol = -np.inf 

523 startax_row = startax_col = stopax_row = stopax_col = None 

524 for ax in axs: 

525 subspec = ax.get_subplotspec() 

526 if subspec.rowspan.start < startrow: 

527 startrow = subspec.rowspan.start 

528 startax_row = ax 

529 if subspec.rowspan.stop > stoprow: 

530 stoprow = subspec.rowspan.stop 

531 stopax_row = ax 

532 if subspec.colspan.start < startcol: 

533 startcol = subspec.colspan.start 

534 startax_col = ax 

535 if subspec.colspan.stop > stopcol: 

536 stopcol = subspec.colspan.stop 

537 stopax_col = ax 

538 return (startrow, stoprow - 1, startax_row, stopax_row, 

539 startcol, stopcol - 1, startax_col, stopax_col)

542def layoutcolorbargridspec(parents, cax, shrink, aspect, location, pad=0.05): 

543 """ 

544 Do the layout for a colorbar, to not overly pollute colorbar.py 

545 

546 *pad* is in fraction of the original axis size. 

547 """ 

548 

549 gs = parents[0].get_subplotspec().get_gridspec() 

550 # parent layout box.... 

551 gslb = gs._layoutbox 

552 

553 lb = layoutbox.LayoutBox(parent=gslb.parent, 

554 name=gslb.parent.name + '.cbar', 

555 artist=cax) 

556 # figure out the row and column extent of the parents. 

557 (minrow, maxrow, minax_row, maxax_row, 

558 mincol, maxcol, minax_col, maxax_col) = _getmaxminrowcolumn(parents) 

559 

560 if location in ('left', 'right'): 

561 lbpos = layoutbox.LayoutBox( 

562 parent=lb, 

563 name=lb.name + '.pos', 

564 tightwidth=False, 

565 pos=True, 

566 subplot=False, 

567 artist=cax) 

568 for ax in parents: 

569 if location == 'right': 

570 order = [ax._layoutbox, lb] 

571 else: 

572 order = [lb, ax._layoutbox] 

573 layoutbox.hstack(order, padding=pad * gslb.width, 

574 strength='strong') 

575 # constrain the height and center... 

576 # This isn't quite right. We'd like the colorbar 

577 # pos to line up w/ the axes poss, not the size of the 

578 # gs. 

579 

580 # Horizontal Layout: need to check all the axes in this gridspec 

581 for ch in gslb.children: 

582 subspec = ch.artist 

583 if location == 'right': 

584 if subspec.colspan.stop - 1 <= maxcol: 

585 order = [subspec._layoutbox, lb] 

586 # arrange to right of the parents 

587 elif subspec.colspan.start > maxcol: 

588 order = [lb, subspec._layoutbox] 

589 elif location == 'left': 

590 if subspec.colspan.start >= mincol: 

591 order = [lb, subspec._layoutbox] 

592 elif subspec.colspan.stop - 1 < mincol: 

593 order = [subspec._layoutbox, lb] 

594 layoutbox.hstack(order, padding=pad * gslb.width, 

595 strength='strong') 

596 

597 # Vertical layout: 

598 maxposlb = minax_row._poslayoutbox 

599 minposlb = maxax_row._poslayoutbox 

600 # now we want the height of the colorbar pos to be 

601 # set by the top and bottom of the min/max axes... 

602 # bottom top 

603 # b t 

604 # h = (top-bottom)*shrink 

605 # b = bottom + (top-bottom - h) / 2. 

606 lbpos.constrain_height( 

607 (maxposlb.top - minposlb.bottom) * 

608 shrink, strength='strong') 

609 lbpos.constrain_bottom( 

610 (maxposlb.top - minposlb.bottom) * 

611 (1 - shrink)/2 + minposlb.bottom, 

612 strength='strong') 

613 

614 # set the width of the pos box 

615 lbpos.constrain_width(lbpos.height * (shrink / aspect), 

616 strength='strong') 

617 elif location in ('bottom', 'top'): 

618 lbpos = layoutbox.LayoutBox( 

619 parent=lb, 

620 name=lb.name + '.pos', 

621 tightheight=True, 

622 pos=True, 

623 subplot=False, 

624 artist=cax) 

625 

626 for ax in parents: 

627 if location == 'bottom': 

628 order = [ax._layoutbox, lb] 

629 else: 

630 order = [lb, ax._layoutbox] 

631 layoutbox.vstack(order, padding=pad * gslb.width, 

632 strength='strong') 

633 

634 # Vertical Layout: need to check all the axes in this gridspec 

635 for ch in gslb.children: 

636 subspec = ch.artist 

637 if location == 'bottom': 

638 if subspec.rowspan.stop - 1 <= minrow: 

639 order = [subspec._layoutbox, lb] 

640 elif subspec.rowspan.start > maxrow: 

641 order = [lb, subspec._layoutbox] 

642 elif location == 'top': 

643 if subspec.rowspan.stop - 1 < minrow: 

644 order = [subspec._layoutbox, lb] 

645 elif subspec.rowspan.start >= maxrow: 

646 order = [lb, subspec._layoutbox] 

647 layoutbox.vstack(order, padding=pad * gslb.width, 

648 strength='strong') 

649 

650 # Do horizontal layout... 

651 maxposlb = maxax_col._poslayoutbox 

652 minposlb = minax_col._poslayoutbox 

653 lbpos.constrain_width((maxposlb.right - minposlb.left) * 

654 shrink) 

655 lbpos.constrain_left( 

656 (maxposlb.right - minposlb.left) * 

657 (1-shrink)/2 + minposlb.left) 

658 # set the height of the pos box 

659 lbpos.constrain_height(lbpos.width * shrink * aspect, 

660 strength='medium') 

661 

662 return lb, lbpos
